def monitor_loop(blueprint_path: str, designer_session_path: str):
    """

            "kpi_data": kpi_result,
            "design_system_status": "PASS" if design_ok else "FAIL",
            "blueprint_ref": blueprint.get("name"),
        }

        print("\n✅ Integrity Monitoring Complete. Final Report:")
